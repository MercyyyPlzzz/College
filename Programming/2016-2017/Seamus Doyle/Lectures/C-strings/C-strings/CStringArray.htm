#include "stdafx.h"

using namespace std;

int _tmain(int argc, _TCHAR* argv[])

{

    char seasons[4][7] = {"Spring",

 “Summer",

 "Autumn",

 "Winter" };

    int monthNumber;

    cout << "Enter month number : ";

    cin >> monthNumber;

 

    switch ( monthNumber )

    {

    case 2:

    case 3:

    case 4:

         cout << "It is " <<seasons[0] << endl;

         break;

    case 5:

    case 6:

    case 7:

         cout << "It is " <<seasons[1] << endl;

         break;

    case 8:

    case 9:

    case 10:

         cout << "It is " << seasons[2] << endl;

         break;

    case 11:

    case 12:

    case 1:

         cout << "It is " << seasons[3] << endl;

         break;

    default:

         cout << "No such month number\n";

    }

    return 0;

}
