void FillList(int list[], int size)

{

}

 

 

int GetLargest(int list[], int size)

{

     int temp = 0;

     for (int i = 0; i < size; i++)

     {

          if (list[i] > temp)

              temp = list[i];

         

     }

     return temp;

}
