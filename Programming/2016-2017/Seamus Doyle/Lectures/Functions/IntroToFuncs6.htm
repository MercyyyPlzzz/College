void DisplayResults(int value)

{

     cout << "Largest value is " << value << endl;

}

 

int GetMax(int first, int second)

{

     if (first > second)

          return first;

     return second;

}

 

void GetInputData(int &first, int &second, int &factor)

{

     cout << "Enter two values :";

     cin >> first >> second;

     cout << "How much do you want them scaled by ? ";

     cin >> factor;

}

 

void ScaleUp(int factor, int &first, int &second)

{

     first *= factor;

     second *= factor;

}
