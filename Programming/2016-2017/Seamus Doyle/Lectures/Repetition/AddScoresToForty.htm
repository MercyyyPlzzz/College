//Program to add scores until total passes 40

#include “stdafx.h”

using namespace std;

int   _tmain(int argc, _TCHAR* argv[])

{

        int           score, total = 0;

       

        //Do some iterations until total passes 40

        do

        {

                  cout << “Enter a score :”;

                  cin >> score;

                  total += score;

        }   while (total < 40 );

 

        cout << “Total of scores = “  <<  total  << endl;

 

}
