int _tmain(int argc, _TCHAR* argv[])

{

        int num1, num2, answer;

        char op;

       

        cout << "Enter calculation (e.g. 5 + 4 ) : ";

        cin >> num1 >> op >> num2;

 

        switch ( op )

        {

        case '+':

                answer = num1 + num2;

                break;

        case '-':

                answer = num1 - num2;

                break;

        case '*':

                answer = num1 * num2;

                break;

        case '/':

                answer = num1 / num2;

                break;

        case '%':

                answer = num1 % num2;

                break;

        default:

                cout << op << "is not an operator\n";

                return 0;

        }

 

 

        cout << setw(10) << num1 << setw(3) << op

 << setw(3) <<num2

<< setw(3) << '='

<< setw(3) << answer << endl;

       

        return 0;

}
