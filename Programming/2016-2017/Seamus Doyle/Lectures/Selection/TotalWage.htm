#include <iostream>

using namespace std;

int main(void)

{

          int           hoursWorked;

          float        hourlyRate;

          float        totalWage;

 

          cout << “Enter number of hours worked :”;

          cin >> hoursWorked;

 

          cout << “Enter hourly rate of pay :”;

          cin >> hourlyRate;

 

          totalWage = hoursWorked * hourlyRate;

 

          if ( hoursWorked > 40 )

          {

                totalWage += ( hoursWorked – 40 ) * hourlyRate * 0.5;

}

 

          cout << “Total wage = “ << totalWage << endl;

 

}
